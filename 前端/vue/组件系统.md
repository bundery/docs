# 组件系统

```html
<div id="app">
    //使用定义的learn-lang组件，把item作为lang属性传递给prop
    <learn-lang v-for="item in langList" v-bind:lang="item">
    </learn-lang>
 </div><

<script>
    //定义一个名称为learn-lang的组件，prop接收一个lang属性
	Vue.component('learn-lang', {
        props: ['lang'],
        template: '<li>{{lang.text}}</li>'
     });

	var app = new Vue({
            el: '#app',
            data: {
                langList: [
                    { text: 'html' },
                    { text: 'css' },
                    { text: 'js' }
                ]
            }
        }) 
</script>
```
